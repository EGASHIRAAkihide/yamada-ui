## インポート

```ts
import { Status } from "@yamada-ui/react"
```

## 使い方

```tsx
<HStack>
  <For each={["info", "success", "warning", "error"]}>
    {(value, index) => <Status key={index} value={value} />}
  </For>
</HStack>
```

### サイズを変更する

```tsx
<VStack>
  <For each={["sm", "md", "lg"]}>
    {(size, index) => (
      <HStack key={index}>
        <For
          each={[
            {
              value: "info",
              text: "Info",
            },
            {
              value: "success",
              text: "Success",
            },
            {
              value: "warning",
              text: "Warning",
            },
            {
              value: "error",
              text: "Error",
            },
          ]}
        >
          {({ value, text }, index) => (
            <Status key={index} size={size} value={value}>
              {text}
            </Status>
          )}
        </For>
      </HStack>
    )}
  </For>
</VStack>
```
