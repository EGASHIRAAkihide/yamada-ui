## Import

```ts
import { CloseButton } from "@yamada-ui/react"
```

## Usage

```tsx functional=true
const items = ["sm", "md", "lg"]

return (
  <Wrap gap="md">
    <For each={items}>
      {(item, index) => <CloseButton size={item} key={index} />}
    </For>
  </Wrap>
)
```
